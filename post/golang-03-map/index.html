	map1 := make(map[string]int)
	s := "I love my work and I love my family too"

	for _, v := range strings.Fields(s, " ") {
		map1[v] += 1
	}

	// 不需要判断，会有默认值赋予；
	for _, v := range strings.Split(s, " ") {
		if _, has := map1[v]; has {
			map1[v] += 1
		} else {
			map1[v] = 1
		}
	}
